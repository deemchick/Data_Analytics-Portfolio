def data_observe(df):
    print(' 10  :')
    if len(df) >= 10:
        display(df.sample(10))
    else:
        display(df)

    print('\n  :')
    df.info()
    
    print('\n    :')
    print(df.duplicated().sum())
   
    print('\n   :')
    display(pd.DataFrame(
                round((df.isna().mean()*100),2), columns=['NaNs, %'])
                .sort_values(by='NaNs, %', ascending=False
            )
        .style.format('{:.2f}')
        .background_gradient('coolwarm')
    )
